Inicio
    Leer peso
    Leer estatura
    Calcular IMC = peso / (estatura * estatura)
    Si IMC < 18.5 entonces
        estado = "Peso bajo"
    Sino si IMC < 25.0 entonces
        estado = "Peso normal"
    Sino si IMC < 30.0 entonces
        estado = "Sobrepeso"
    Sino si IMC < 40.0 entonces
        estado = "Obesidad"
    Sino
        estado = "Obesidad extrema"
    FinSi
    Imprimir IMC
    Imprimir estado
Fin
